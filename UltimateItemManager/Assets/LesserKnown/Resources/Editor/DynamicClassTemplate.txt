using System;
using System.Collections.Generic;
using UnityEngine;

[CreateAssetMenu(fileName ="DynamicClass", menuName = "Lesser/Dynamic")]
public class DynamicData
{	
	private static DynamicData instance;
	public static DynamicData Instance
	{
		get
		{
			if(instance == null)
			{
				instance = new DynamicData();
			}
			return instance;
			}
	}	
	public class ItemsClass
	{
		>:FOREVERY:<
		public [0] Name;
		>:ENDFOREVERY:<	
		
		public ItemsClass
		(
		>:FOREVERY Param:<
		[0] Name
		>:ENDFOREVERY Param:<	
		) 
		{
		>:FOREVERY Assign:<
		this.Name = Name;
		>:ENDFOREVERY Assign:<	
		}
	}
	>:FOREVERY Class:<
	public ItemsClass >:className:< = new ItemsClass
	(
	>:FILL_PARAM:<
	);
	>:ENDFOREVERY Class:<	
}
